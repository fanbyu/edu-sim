```python
from abc import ABC, abstractmethod
from typing import Dict, List, Any
import pandas as pd

class StructuredDataSource(ABC):
    """结构化数据源抽象基类"""
    
    @abstractmethod
    def load(self, source_path: str) -> Dict[str, Any]:
        pass
    
    @abstractmethod
    def validate(self, data: Dict[str, Any]) -> bool:
        pass

class ExamDataLoader(StructuredDataSource):
    """考试数据加载器 - 原生支持教育场景"""
    
    def load(self, source_path: str) -> Dict[str, Any]:
        """
        加载考试数据,自动识别:
        - 学生元数据 (student_id, class, demographics)
        - 试题信息 (item_id, difficulty, concepts)
        - 作答记录 (student_id, item_id, score, time_spent)
        """
        df_responses = pd.read_csv(f"{source_path}/responses.csv")
        df_students = pd.read_csv(f"{source_path}/students.csv")
        df_items = pd.read_csv(f"{source_path}/items.csv")
        
        return {
            "type": "exam",
            "students": df_students.to_dict('records'),
            "items": df_items.to_dict('records'),
            "responses": df_responses.to_dict('records')
        }
    
    def validate(self, data: Dict[str, Any]) -> bool:
        """验证数据完整性"""
        required_keys = ['students', 'items', 'responses']
        return all(key in data for key in required_keys)

class LearningPathLoader(StructuredDataSource):
    """学习路径数据加载器"""
    pass
```

### 3. **Agent 建模层 (Agent Modeling Layer)**

```python
# core/agent_modeling/student_agent.py

from dataclasses import dataclass
from typing import List

@dataclass
class StudentAgent:
    """学生 Agent - 原生支持 IRT 参数"""
    
    student_id: str
    cognitive_level: float  # θ (IRT theta)
    learning_style: str     # visual/auditory/kinesthetic
    anxiety_threshold: float
    motivation_level: float
    
    # 动态属性
    mastered_concepts: List[str]
    recent_performance: List[float]
    
    def predict_response_probability(self, item_difficulty: float) -> float:
        """基于 IRT 2PL 模型预测答对概率"""
        import numpy as np
        discrimination = 1.0  # 可配置
        return 1 / (1 + np.exp(-discrimination * (self.cognitive_level - item_difficulty)))
    
    def update_after_intervention(self, intervention_type: str):
        """干预后更新状态"""
        if intervention_type == "heuristic":
            self.anxiety_threshold -= 0.1
            self.motivation_level += 0.15
        elif intervention_type == "direct_instruction":
            self.cognitive_level += 0.2
```

### 4. **IRT 校准引擎 (IRT Calibration Engine)**

```python
# core/agent_modeling/irt_engine.py

import numpy as np
from scipy.optimize import minimize

class IRTEngine:
    """IRT 参数校准引擎 - 原生集成"""
    
    def __init__(self, model_type: str = "2PL"):
        self.model_type = model_type
    
    def calibrate(self, response_matrix: np.ndarray) -> Dict:
        """
        校准学生和题目参数
        
        Args:
            response_matrix: N×M 矩阵,N=学生数,M=题目数
            
        Returns:
            {'student_thetas': [...], 'item_difficulties': [...]}
        """
        n_students, n_items = response_matrix.shape
        
        # 联合极大似然估计 (JMLE)
        def neg_log_likelihood(params):
            thetas = params[:n_students]
            bs = params[n_students:n_students+n_items]
            
            log_lik = 0
            for i in range(n_students):
                for j in range(n_items):
                    if not np.isnan(response_matrix[i, j]):
                        p = self._irt_prob(thetas[i], bs[j])
                        p = np.clip(p, 1e-6, 1-1e-6)
                        log_lik += response_matrix[i,j] * np.log(p) + \
                                  (1-response_matrix[i,j]) * np.log(1-p)
            return -log_lik
        
        initial_params = np.zeros(n_students + n_items)
        result = minimize(neg_log_likelihood, initial_params, method='L-BFGS-B')
        
        return {
            'student_thetas': result.x[:n_students].tolist(),
            'item_difficulties': result.x[n_students:].tolist()
        }
    
    def _irt_prob(self, theta: float, b: float, a: float = 1.0) -> float:
        """IRT 2PL 模型"""
        return 1 / (1 + np.exp(-a * (theta - b)))
```
